a bash script for running claude code in a restricted user environment

claude-agent.sh bash 
#!/bin/bash
# --- CONFIGURATION ---
AGENT_USER="claude-agent"
SHARE_GROUP="claude-share"
AGENT_HOME="/home/$AGENT_USER"
CLAUDE_BIN="$AGENT_HOME/.local/bin/claude"

# --- HELPER: CHECK SUDO ---
check_sudo() {
    if [ "$EUID" -ne 0 ]; then
        echo "Please run with sudo (e.g., sudo $0 $1)"
        exit 1
    fi
}

case "$1" in
    setup)
        echo "[+] Creating restricted environment..."
        check_sudo
        groupadd -f "$SHARE_GROUP"
        useradd -m -s /bin/bash -g "$SHARE_GROUP" "$AGENT_USER"
        usermod -aG "$SHARE_GROUP" "$SUDO_USER"
        chmod 700 "$AGENT_HOME"
        echo "Done! Restart your terminal/session for group changes to take effect."
        ;;
    allow)
        TARGET_DIR="${2:-$(pwd)}"
        echo "[+] Granting $AGENT_USER access to: $TARGET_DIR"
        check_sudo
        chgrp -R "$SHARE_GROUP" "$TARGET_DIR"
        chmod -R g+rw "$TARGET_DIR"
        find "$TARGET_DIR" -type d -exec chmod g+x {} +
        chmod g+s "$TARGET_DIR"
        echo "Access granted."
        ;;
    run)
        # Check if we have an API key to pass along
        ENV_VARS=""
        [[ ! -z "$ANTHROPIC_API_KEY" ]] && ENV_VARS="ANTHROPIC_API_KEY=$ANTHROPIC_API_KEY"

        echo "[+] Dropping into $AGENT_USER shell at $(pwd)..."
        # We use -i (login shell) to ensure the agent's PATH includes its local bin
        sudo -u "$AGENT_USER" $ENV_VARS bash -c "cd $(pwd) && exec bash --login"
        ;;
    *)
        echo "Usage: $0 {setup|install|allow|run}"
        echo "  setup   : Create the user and group (Run once as root)"
        echo "  install : Download and install Claude (Run as yourself)"
        echo "  allow   : Grant Claude access to current folder (Run as root)"
        echo "  run     : Start the restricted session (Run as yourself)"
        ;;
esac
